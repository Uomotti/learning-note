# five - mistake

# 1. Reactivity Caveat 1

axios로 데이터를 받아오고 나서 obj.newItem 형식으로 데이터를 추가했을 때 화면에 반응성이 없는 사례

<div id="app">
  <div>
    <h1>User Info</h1>
    <p>Name : {{ user.name }}</p>
    <p>Email : {{ user.email }}</p>
    <p>Region : {{ user.region }}</p>
    <button @click="changeName">change Name</button>
    <button @click="changeRegion">change Region</button>
  </div>
  <div>
    <h1>User Data</h1>
    <div>
      {{ user }}
    </div>
  </div>
</div>

new Vue({
  el: "#app",
  data: {
    user: {},
  },
  created() {
    this.fetchUsers();
  },
  methods: {
    fetchUsers() {
      fetch("https://jsonplaceholder.typicode.com/users/1")
        .then(response => response.json())
        .then(data => {
          this.user = data;
          // this.user.region = "Pangyo";  추가로 객체 key가 설정된 경우 reactivity 작동오류
          this.$set(this.user, "region", "Pangyo"); // $set을 이용해 재정의
        })
        .catch(error => console.log(error));
    },
    changeName() {
      this.user.name = "Josh";
    },
    changeRegion() {
      this.user.region = "Seoul";
      console.log("The region has been changed", this.user);
    },
  },
});

# 2. Reactivity Caveat 2

state의 객체에 새로 property를 추가하거나 delete로 지우는 경우 화면이 갱신 되지 않는다(반응성이 없다)

<div id="app">
  <div>
    <h1>User Info</h1>
    <p>Name : {{ this.$store.state.user.name }}</p>
    <p>Email : {{ this.$store.state.user.email }}</p>
    <button @click="removeName">Remove Name</button>
    <button @click="addNewProperty">Add New Property</button>
  </div>
  <div>
    <h1>User Data</h1>
    <div>
      {{ this.$store.state.user }}
    </div>
  </div>
</div>

const store = new Vuex.Store({
  state: {
    user: {},
  },
  mutations: {
    setUser(state, user) {
      state.user = user;
    },
    deleteName(state) {
      delete state.user.name;
      // state.user.name = '';
    },
    addNum(state) {
      state.user.num = 300;
      // API 형식 - Vue.set(object, key, value)
      // Vue.set(state.user, 'num', 300);
    },
  },
  actions: {
    fetchUser({ commit }) {
      fetch("https://jsonplaceholder.typicode.com/users/1")
        .then(response => response.json())
        .then(data => commit("setUser", data))
        .catch(error => console.log(error));
    },
  },
});

new Vue({
  store,
  el: "#app",
  created() {
    this.$store.dispatch("fetchUser");
  },
  methods: {
    removeName() {
      this.$store.commit("deleteName");
      console.log("The name has been removed", this.$store.state.user);
    },
    addNewProperty() {
      this.$store.commit("addNum");
      console.log("A number has been added", this.$store.state.user);
    },
  },
});

# 3. DOM Direct Accessing

v-for문과 ref 를 이용해 코드를 더 깔끔하게 짤 수 있다

<div id="app">
  <ul>
    <!-- before -->
    <!-- <li @click="removeItem">
        <span>메뉴 2</span>
        <div class="child hide">
          메뉴 설명
        </div>
      </li>
      <li @click="removeItem">
        <span>메뉴 3</span>
        <div class="child hide">
          메뉴 설명
        </div>
      </li> -->

    <!-- After -->
    <li v-for="( item, index ) in items" @click="removeItem(index)">
      <span>{{ item }}</span>
      <div class="child hide" ref="listItem">
        메뉴 설명
      </div>
    </li>
  </ul>
</div>

new Vue({
  el: "#app",
  data: {
    items: ["메뉴 1", "메뉴 2", "메뉴 3"],
  },
  mounted() {
    console.log(this.$refs);
  },
  methods: {
    removeItem(index) {
      this.$refs.listItem[index].classList.toggle("hide");
      // event.target.lastChild.classList.toggle("hide");
      // this.$refs[`listItem${index}`][0].classList.toggle("hide");
    },
  },
});

# 5. Undefined Ref

<div id="app">
  <section>
    <h1>Finding Ref property</h1>
  </section>
  <template v-if="loading">
    <div>
      Displayed when loading
    </div>
  </template>
  <template v-if="loading">
    <div ref="item" @click="refConsole">
      Hello Ref
    </div>
  </template>
</div>

new Vue({
  el: "#app",
  data: {
    loading: true,
  },
  mounted() {
    this.$refs.item.style = "color: blue";
  },
  methods: {
    refConsole() {
      console.log(this.$refs.item);
    },
  },
});
