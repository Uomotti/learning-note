# 컴포넌트 통신

# $emit

# 하위 컴포넌트에서 이벤트 emit 셋팅

<button v-on:click="eventName"></button>

methods:{
    eventName : function(){
        this.$emit("emit-name",인자1,인자2, ... );
    }
}

# 상위 컴포넌트에서 이벤트 받기

<event-emit-component v-on:emit-name="eventMethods"></event-emit-component>

methods:{
    eventMethods:function(){
        alert("이벤트 실행");
    }
}
