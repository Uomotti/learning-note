# async & await

# 이전 콜백 처리 방법

fetchUser() {
    let vm = this;
    axios(url1)
        .then(response => {
            if (response.data.id == 1) {
                axios(url2)
                    .then(response2 => {
                        // console.log(this.$refs.textBox);
                        this.textBox = response2.data;
                    })
                    .catch(error => {
                        console.log(error);
                    });
            }
        })
        .catch(error => {});
},

# async & await 사용법

async login() {
    try {
        // 1. 사용자 정보를 요청
        // 2. 사용자 정보의 id를 체크
        // 3. 할 일 정보를 요청
        var { data } = await this.fetchUser();
        if (data.id === 1) {
            var response = await this.fetchTodo();
            console.log(response.data);
        }
    } catch (error) {
        console.log(error);
    } finally {
        console.log('complete');
    }
},
  1. 비동기 처리를 포함하는 메소드 함수에 async를 붙여준다.
  2. 메소드 함수 내에 비동기 처리 호출 함수에 await를 붙여준다
  3. try 와 catch로 데이터 처리를 한다
