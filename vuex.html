# vuex

# vuex 설치

npm i vuex

# vuex 등록

// store.js
import Vue from "vue";
import Vuex from "vuex";

Vue.use(Vuex);

export const store = new Vuex.Store({
  // ..
});

// main.js
import Vue from "vue";
import { store } from "./store.js";

new Vue({
  store: store,
});

# vuex 사용법

# state

state 는 data 와 같다

new Vuex.Store({
  state: {
    message: "Hello Vue.js",
  },
});

# state 접근방법

<div>{{ this.$store.state.message }}</div>

# mutations

mutations 는 methods와 같다 뮤테이션(mutations)은 뷰엑스에서 상태 값을 변경하는 유일한 방법입니다. 상태는 항상 뮤테이션으로 변경됩니다.

new Vuex.Store({
  state: {
    message: "Hello Vue.js",
  },
  mutations: {
    reverseMessage(state) {
      state.message = state.message
        .split("")
        .reverse()
        .join("");
    },
  },
});

# mutations 접근방법

new Vue({
  methods: {
    reverseMsg() {
      this.$store.commit("reverseMessage");
    },
  },
});

# getters

getters 는 computed 와 같다. 뷰엑스에서 상태 값이 변경되면 실행되는 함수다

new Vuex.Store({
  state: {
    message: "Hello Vue.js",
  },
  getters: {
    reverseMessage(state) {
      return state.message
        .split("")
        .reverse()
        .join("");
    },
  },
});

# getters 접근방법

<div>{{ this.$store.getters.reverseMessage }}</div>
